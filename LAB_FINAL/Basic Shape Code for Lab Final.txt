#include <windows.h>
#include <GL/gl.h>
#include <GL/glut.h>
#include <string.h>
#include <math.h>
// Function to draw text on screen
void drawText(float x, float y, const char* text) {
glRasterPos2f(x, y);
for (int i = 0; i < strlen(text); i++) {
glutBitmapCharacter(GLUT_BITMAP_HELVETICA_12, text[i]);
}
}

//Draw Heading
void drawHext(float x, float y, const char* text) {
glRasterPos2f(x, y);
for (int i = 0; i < strlen(text); i++) {
glutBitmapCharacter(GLUT_BITMAP_HELVETICA_18, text[i]);
}
}

// Function to draw a circle
void drawCircle(float cx, float cy, float radius)
{
    glBegin(GL_POLYGON);
    for (int i = 0; i < 360; i++)
    {
        float theta = i * (M_PI / 180.0f);
        float x = radius * cos(theta) + cx;
        float y = radius * sin(theta) + cy;
        glVertex2f(x, y);
    }
    glEnd();
}

// Display callback function to render the shapes
void display(void)
{
glClear(GL_COLOR_BUFFER_BIT);

//glBegin(GL_LINES); //Draw single line
//glBegin(GL_LINE_LOOP);  //Draw borders of any shapes

//Heading
glColor3f(0.0f, 0.0f, 0.0f); // Red
drawHext(40, 92, "ALL GEOMARTRIC SHAPES");

// Triangle
glColor3f(0.9f, 0.1f, 0.1f); // Red
glBegin(GL_TRIANGLES);
glVertex2f(5, 70);
glVertex2f(25, 70);
glVertex2f(15, 87);
glEnd();
drawText(13, 65, "3 edges");

// Parallelogram
glColor3f(0.0f, 0.8f, 0.4f); // Teal
glBegin(GL_QUADS);
glVertex2f(30, 70);
glVertex2f(45, 70);
glVertex2f(50, 87);
glVertex2f(35, 87);
glEnd();
drawText(35, 65, "4 edges");

// Square
glColor3f(0.0f, 0.4f, 0.9f); // Blue
glBegin(GL_QUADS);
glVertex2f(55, 70);
glVertex2f(70, 70);
glVertex2f(70, 87);
glVertex2f(55, 87);
glEnd();
drawText(60, 65, "4 edges");


glEnd();

// Circle
glColor3f(0.9f, 0.2f, 1.0f);
drawCircle(85, 78, 8);
glEnd();
drawText(83, 6, "Circle");

glFlush();
}

// Initialization
void init(void)
{
glClearColor(1.0, 1.0, 1.0, 1.0); //Background Color
glMatrixMode(GL_PROJECTION);
glLoadIdentity();
glOrtho(0.0, 100.0, 0.0, 100.0, -1.0, 1.0);
}

int main(int argc, char** argv)
{

glutInit(&argc, argv);
glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
glutInitWindowSize(1000, 1000); //Display Size
glutInitWindowPosition(0,0);
glutCreateWindow("Basic Shapes in OpenGL (Using glVertex3f)");  //Title
init();
glutDisplayFunc(display);
glutMainLoop();
return 0;
}
